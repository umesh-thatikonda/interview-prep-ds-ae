def rearrange_digits(number: int) -> int: 
    # Convert the number to a string and take the absolute value 
    # (even though the input is guaranteed to be positive as per the problem statement) 
    # Sort the digits in descending order 
    sorted_digits = sorted( str(abs(number)), reverse=True) 

    # Join the sorted digits into a single string 
    res = “”.join(sorted_digits) 

    # Print the intermediate result (for debugging purposes) 
    print(res) 

    # Convert the string back to an integer and return it 
    return int(res)

from typing import List 

def largest_number_of_people(meetings: List[Meeting]) -> int: 
    # Initialize a dictionary to track the total audience at each hour 
    time_limit = {i: 0 for i in range(1, 25)} 

 
 
 
 
 
 
 
 
 
 
 
     
    # Iterate through each meeting 
    for item in meetings: 
        start_time = item.start 
        end_time = item.end 

        # Increment the audience count for each hour the meeting is scheduled 
        for i in range(start_time, end_time + 1): 
            time_limit[i] += item.audience 

    # Find the maximum audience count across all hours 
    max_value = max(time_limit.values()) 

    return max_value 

[Set 2] The bookstore hosts a set of yearly workshops. 
Each workshop repeats the same classes-each year. 
Given a  list-of-workshops,  return-the-largest number of classes that were hosted in total across 
any 2 consecutive-years. 

Example:   
[Workshop(name = "Intro to archival science", classes per year = 4, Start = 2000，end = 2003), 
 Workshop(name = "Bind your own book", classes per year = 1, start =  2001, end = 2004), 
 Workshop(name = "Monthly discussion circle", classes per year = 12, start = 2004, end=2006), 

Result:  25  

Explanation:  The largest  number of classes across 2 consecutive years were hosted between  
2004 (13-classes)  and  2005 (12 classes) when the last two workshops were taking place at the 
same time. 

from typing import List 
from dataclasses import dataclass 

@dataclass 
class Workshop: 
    name: str 
    classes_per_year: int 
    start: int  
    end: int 

def largest_number_of_classes(workshops: List[Workshop]) -> int: 
    # ------------------------------------------------------------------ 
    # 1) Build the year → total-classes map 
    # ------------------------------------------------------------------ 

 
         
     
     
  
 
 
 
 
 
 
    year_classes = {}                         # {year: total classes} 

    for workshop in workshops: 
        for year in range(workshop.start, workshop.end + 1): 
            if year in year_classes: 
                year_classes[year] += workshop.classes_per_year 
            else: 
                year_classes[year] = workshop.classes_per_year 

    # ------------------------------------------------------------------ 
    # 2) Fill the gaps: make sure every year between min and max exists 
    # ------------------------------------------------------------------ 
    if year_classes:                          # avoid min()/max() on empty dict 
        first_year = min(year_classes)        # earliest year we saw 
        last_year  = max(year_classes)        # latest  year we saw 

        for yr in range(first_year, last_year + 1): 
            if yr not in year_classes:        # impute missing year with 0 
                year_classes[yr] = 0 

    # ------------------------------------------------------------------ 
    # 3) Scan consecutive pairs and keep the best sum 
    # ------------------------------------------------------------------ 
    max_classes = 0 
    years = sorted(year_classes.keys())       # now a *continuous* list 

    for i in range(len(years) - 1): 
        if years[i + 1] - years[i] == 1:      # guaranteed true after filling, but kept for clarity 
            current_two_years = (year_classes[years[i]] + 
                                 year_classes[years[i + 1]]) 
            if current_two_years > max_classes: 
                max_classes = current_two_years 

    return max_classes

from typing import List 

def find_editions(books: List[str]) -> List[str]: 
    if not books: 
        return [] 

    # ❶ Sort once → enables linear prefix grouping 
    books_sorted = sorted(books) 

    n = len(books_sorted) 
    editions: List[str] = [] 
    i = 0                                    # left pointer – start of current block 

    while i < n: 
        original = books_sorted[i] 
        j = i + 1                            # right pointer – scans possible editions 

        # ❷ Collect consecutive strings that start with "original + space" 
        prefix = original + " " 
        while j < n and books_sorted[j].startswith(prefix): 
            editions.append(books_sorted[j]) 
            j += 1 

        # ❸ Move left pointer to the next potential original 
        i = j                                # skip over the block just processed 

    return editions

from typing import List 

def can_fit_books(bookshelf_sizes: List[int], bookshelf_titles: List[str]) -> bool: 
    """ 
    Determines if all book titles can fit into the given bookshelf sizes. 

    Args: 
    bookshelf_sizes (List[int]): A list of available bookshelf slot sizes. 
    bookshelf_titles (List[str]): A list of book titles. 

    Returns: 
    bool: True if all books can fit, otherwise False. 
    """ 
    # Ensure there are enough slots for all titles 
    if len(bookshelf_titles) > len(bookshelf_sizes): 
        return False 

    # Sort bookshelf sizes in ascending order 
    bookshelf_sizes.sort(reverse=True) 

    # Sort book titles by length in ascending order 
    bookshelf_titles.sort(key=len, reverse=True) 

    # Try to fit each book into a shelf 
    for i in range(len(bookshelf_titles)): 
        if len(bookshelf_titles[i]) > bookshelf_sizes[i]: 
            return False  # If a title doesn't fit, return False 

    return True  # All books fit successfully

from typing import List 

def first_missing_positive(nums: List[int]) -> int: 
    n = len(nums) 

    # First step is to mark numbers (num < 0) and (num > n) with a special marker number 
(n+1) 
    # (we can ignore those because the missing positive integer must be in the range 1 to 
n+1) 
    for i in range(n): 
        if nums[i] <= 0 or nums[i] > n: 
            nums[i] = n + 1 

    # Second step is to use the index as a hash key and the number sign as a presence 
detector 
    for i in range(n): 
        num = abs(nums[i]) 
        if num <= n: 
            nums[num - 1] = -abs(nums[num - 1]) 

    # Third step is to find the first cell which isn't negative (i.e. has a positive value) 
    for i in range(n): 
        if nums[i] > 0: 
            return i + 1 

    return n + 1 

# Example Usage 
print(first_missing_positive([3, 4, -1, 1]))  # Output: 2 
print(first_missing_positive([1, 2, 0]))      # Output: 3 

Time Complexity: O(n) 
Space Complexity: O(1)

Approach 1 (dictionary by category): 

from typing import List, Tuple 

def get_max_score(books: List[Tuple[str, int]]) -> int: 
    category_best = {} 
    for category, score in books: 
        if category not in category_best or score > category_best[category]: 
            category_best[category] = score 
       else :  

category_best[category] = max(category_best[category], score) 

    return sum(sorted(category_best.values(), reverse=True)[:3]) 

Approach 2 (group by category, sort + slice): 

from typing import List, Tuple 

def get_max_score(books: List[Tuple[str, int]]) -> int: 
    scores_by_category = {} 
    for cat, score in books: 
        scores_by_category.setdefault(cat, []).append(score) 
    top_scores = [max(v) for v in scores_by_category.values()] 
    return sum(sorted(top_scores, reverse=True)[:3])

Q2) Every time a book is checked out or returned at the library, a log entry is created with 
the ID of the book and a flag that is true for checkouts or false for returns. 

Given an ordered list of log entries, write a function that returns false if: 

* a book was checked out when it was already checked out 

* a book was returned when it was not checked out. 

Assume no books are checked out before the first log entry in the input. 

Approach (set tracking): 

from dataclasses import dataclass 

@dataclass 
class LogEntry: 
    book_id: int 
    is_checkout: bool 

def are_log_entries_valid(log_entries: list[LogEntry]) -> bool: 
    checked_out = set() 
    for entry in log_entries: 
        if entry.is_checkout: 
            if entry.book_id in checked_out: 
                return False 
            checked_out.add(entry.book_id) 
        else: 
            if entry.book_id not in checked_out: 
                return False 
            checked_out.remove(entry.book_id) 

 
 
 
 
 
 
 
 
 
    return True

4) Given two words, source and target, determine whether target can be formed by 
deleting characters from source without re‑ordering the remaining letters.  If impossible, 
return None. 

Give the total count of deleted characters (duplicates counted). 

Approach 1 : 

def deletions_total(src: str, tgt: str): 
    """Total characters to delete so that src → tgt, else None.""" 
    if src is None or tgt is None: 
        return None 
    i = j = 0 
    deletions = 0 
    while i < len(src) and j < len(tgt): 
        if src[i] == tgt[j]: 
            j += 1            # keep char 
        else: 
            deletions += 1     # delete char 

 
 
 
 
 
 
 
        i += 1 
    deletions += len(src) - i  # tail deletions 
    return deletions if j == len(tgt) else None

Approach 2 : 

def deletions_distinct(src: str, tgt: str): 
    """Distinct letters to delete so that src → tgt, else None.""" 
    if src is None or tgt is None: 
        return None 
    i = j = 0 
    removed = set() 
    while i < len(src) and j < len(tgt): 
        if src[i] == tgt[j]: 
            j += 1 
        else: 
            removed.add(src[i]) 
        i += 1 
    if j != len(tgt): 
        return None 
    removed.update(src[i:])      # remaining chars are deletions 
    return len(removed)
